````bash
#!/bin/bash

test -z "$BASH" && return

# don't put duplicate lines in the history. See bash(1) for more options
# ... or force ignoredups and ignorespace
HISTCONTROL=ignoredups:ignorespace:erasedups
HISTTIMEFORMAT="%F %T: "

# for setting history length see HISTSIZE and HISTFILESIZE in bash(1)
HISTSIZE=30000
HISTFILESIZE=30000

# put commmand in history file before executing it
PROMPT_COMMAND="history -a"

export PAGER=most

shopt -s cdspell
shopt -s autocd

alias vi=vim
alias igrep='grep -i --color=auto'
alias ccat='pygmentize2 -O style=monokai -f console256 -g'

function prompt ()
{
  case "$1" in
    hsw)
      PS1='\h[$SHLVL]: \w \$ '
      ;;
    hw)
      PS1='\h: \w \$ '
      ;;
    uh)
      PS1='\u@\h\$ '
      ;;
    uhsHw)
      PS1='\u@\h[$SHLVL]:\#: \w \$ '
      ;;
    uhsw)
      PS1='\u@\h[$SHLVL]: \w \$ '
      ;;
    uhw)
      PS1='\u@\h: \w \$ '
      ;;
    p)
      PS1="\n\[\e[0;36m\]\t \[\e[1;30m\][\[\e[1;34m\]\u@\H\[\e[1;30m\]:\[\e[0;37m\]${SSH_TTY:-o} \[\e[0;32m\]+${SHLVL}\[\e[1;30m\]] \[\e[01;37m\]\w\[\e[00m\] \n\$ "
      ;;
  esac
}

if [ -f ~/.bash_aliases ]; then
    . ~/.bash_aliases
fi
````
